*/

/**
* The className of this class.
*
* @type {string}
* @protected
*/
b._classname = $_CLASSNAME;

* The 'dirty' flag of this object.
*
* @type {boolean}

* @type {number}
* @protected
*/

* @const
*/

* The size of each item.

*/
